// Demonstrating string concatenation (+), substring (SUBSTR), and both together
"Welcome " + "to " + "Programming!"                  // Basic string concatenation
"Happy " + "Coding " + "Day!"                        // Multiple strings concatenation
"Answer is: " + "100"                                // Concatenating string with numeric string
substr("Good Afternoon" 5 9)                         // Extracting substring
"The value is: " + "3.1416"                          // String concatenation with a numeric string
"A" + "I" + " is trending"                           // Concatenating strings without spaces
"Joining " + "parts" + " of " + "text"               // Concatenating with spaces
"Hi " + "there, " + "what's " + "up?"                // Greeting concatenation
"Combining " + "strings " + "is " + "cool!"          // Positive statement concatenation
substr(("AB" + "12") 2 3)                            // Substring of a concatenated string
"The lazy " + "dog " + "jumps " + "over"             // Concatenation of a phrase
"Learning " + "string " + "operations"               // Concatenation in a learning context
"hello" + substr("hello world" 6 5)                  // Concatenating string with a substring
len("Consistency") + " is key"                       // Concatenating result of len() with string
substr("hello world" len("hello ") len("world"))     // Nested functions to extract substring
